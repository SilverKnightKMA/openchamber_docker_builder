Apt package set check
Command:
  python3 - <<'PY'
  from pathlib import Path
  text = Path('Dockerfile.dockerfile').read_text().splitlines()
  packages = []
  in_block = False
  for line in text:
      stripped = line.strip()
      if stripped.startswith('RUN apt-get update && apt-get install'):
          in_block = True
          continue
      if in_block:
          if stripped.startswith('&& '):
              break
          if stripped.endswith('\\'):
              packages.append(stripped[:-1].strip())
          elif stripped:
              packages.append(stripped)
  approved = 'bash ca-certificates curl git openssh-client python3 python3-pip python3-venv tar unzip xz-utils file jq less procps psmisc iproute2 iptables kmod netcat-openbsd lsof rsync sudo dnsutils iputils-ping nano git-lfs build-essential'.split()
  print('apt packages:', ' '.join(packages))
  print('missing approved:', ' '.join(sorted(set(approved) - set(packages))) or 'none')
  print('extra packages:', ' '.join(sorted(set(packages) - set(approved))) or 'none')
  PY
